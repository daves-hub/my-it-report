\section*{Abbreviations}

\begin{itemize}
\item \textbf{SIWES} – Students Industrial Work Experience Scheme
\item \textbf{UI/UX} – User Interface / User Experience
\item \textbf{LaTeX} – A document preparation and typesetting system
\item \textbf{Figma} – A collaborative UI/UX design tool
\item \textbf{WPM} – Words Per Minute
\item \textbf{SVG} – Scalable Vector Graphics
\item \textbf{PDF} – Portable Document Format
\item \textbf{HTML} – HyperText Markup Language
\item \textbf{CSS} – Cascading Style Sheets
\item \textbf{RGB} – Red, Green, Blue (color model)
\item \textbf{amsmath} – A LaTeX package for mathematical typesetting
\item \textbf{biblatex} – A LaTeX package for bibliography management
\item \textbf{enumitem} – A LaTeX package for list customization
\item \textbf{multicol} – A LaTeX package for multiple column layouts
\item \textbf{geometry} – A LaTeX package for page layout control
\end{itemize}
